## Architecture

### Repository Pattern

```rust
// src/repository/portfolio_repository.rs
use async_trait::async_trait;
use sqlx::PgPool;
use uuid::Uuid;

#[async_trait]
pub trait PortfolioRepository: Send + Sync {
    async fn create(&self, portfolio: &Portfolio) -> Result<Portfolio, sqlx::Error>;
    async fn find_by_id(&self, id: Uuid) -> Result<Option<Portfolio>, sqlx::Error>;
    async fn find_by_user_id(&self, user_id: &str) -> Result<Vec<Portfolio>, sqlx::Error>;
    async fn update(&self, portfolio: &Portfolio) -> Result<Portfolio, sqlx::Error>;
    async fn delete(&self, id: Uuid) -> Result<bool, sqlx::Error>;
}

pub struct PostgresPortfolioRepository {
    pool: PgPool,
}

impl PostgresPortfolioRepository {
    pub fn new(pool: PgPool) -> Self {
        Self { pool }
    }
}

#[async_trait]
impl PortfolioRepository for PostgresPortfolioRepository {
    async fn create(&self, portfolio: &Portfolio) -> Result<Portfolio, sqlx::Error> {
        sqlx::query_as!(
            Portfolio,
            r#"
            INSERT INTO portfolios (id, user_id, name, description, total_value, created_at, updated_at)
            VALUES ($1, $2, $3, $4, $5, $6, $7)
            RETURNING *
            "#,
            portfolio.id,
            portfolio.user_id,
            portfolio.name,
            portfolio.description,
            portfolio.total_value,
            portfolio.created_at,
            portfolio.updated_at
        )
        .fetch_one(&self.pool)
        .await
    }
    
    // ... other methods
}
```

### Service Layer

```rust
// src/service/portfolio_service.rs
use crate::model::{Portfolio, CreatePortfolioRequest};
use crate::repository::PortfolioRepository;
use uuid::Uuid;
use chrono::Utc;

pub struct PortfolioService<R: PortfolioRepository> {
    repository: R,
}

impl<R: PortfolioRepository> PortfolioService<R> {
    pub fn new(repository: R) -> Self {
        Self { repository }
    }
    
    pub async fn create_portfolio(
        &self,
        user_id: String,
        req: CreatePortfolioRequest,
    ) -> Result<Portfolio, Box<dyn std::error::Error>> {
        let portfolio = Portfolio {
            id: Uuid::new_v4(),
            user_id,
            name: req.name,
            description: req.description,
            total_value: 0.0,
            created_at: Utc::now(),
            updated_at: Utc::now(),
        };
        
        let created = self.repository.create(&portfolio).await?;
        Ok(created)
    }
    
    // ... other methods
}
```

### HTTP Handlers

```rust
// src/handler/portfolio_handler.rs
use actix_web::{web, HttpResponse, Result};
use crate::service::PortfolioService;
use crate::model::CreatePortfolioRequest;

pub async fn create_portfolio(
    service: web::Data<PortfolioService>,
    req: web::Json<CreatePortfolioRequest>,
    user_id: web::ReqData<String>,
) -> Result<HttpResponse> {
    match service.create_portfolio(user_id.into_inner(), req.into_inner()).await {
        Ok(portfolio) => Ok(HttpResponse::Created().json(portfolio)),
        Err(e) => Ok(HttpResponse::InternalServerError().json(json!({
            "error": e.to_string()
        }))),
    }
}

pub async fn list_portfolios(
    service: web::Data<PortfolioService>,
    user_id: web::ReqData<String>,
) -> Result<HttpResponse> {
    match service.list_portfolios(user_id.into_inner()).await {
        Ok(portfolios) => Ok(HttpResponse::Ok().json(portfolios)),
        Err(e) => Ok(HttpResponse::InternalServerError().json(json!({
            "error": e.to_string()
        }))),
    }
}
```

## Authentication

### JWT Middleware

```rust
// src/middleware/auth.rs
use actix_web::{dev::ServiceRequest, Error, HttpMessage};
use actix_web_httpauth::extractors::bearer::BearerAuth;

pub async fn validator(
    req: ServiceRequest,
    credentials: BearerAuth,
) -> Result<ServiceRequest, Error> {
    let token = credentials.token();
    
    // Validate token with OIDC provider
    match validate_token(token).await {
        Ok(claims) => {
            req.extensions_mut().insert(claims.sub);
            Ok(req)
        }
        Err(_) => Err(actix_web::error::ErrorUnauthorized("Invalid token")),
    }
}
```
